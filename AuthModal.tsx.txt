import React, { useState } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { 
  X, Mail, Lock, User, Store, ArrowRight, ShieldCheck, 
  Store as StoreIcon, Package, User as UserIcon, LogIn, ChevronRight,
  Briefcase
} from 'lucide-react';
import { useMarketplace } from '../context/MarketplaceContext';
import { auth, db } from '../lib/firebase';
import { 
  signInWithPopup, 
  GoogleAuthProvider, 
  createUserWithEmailAndPassword, 
  signInWithEmailAndPassword 
} from 'firebase/auth';
import { doc, setDoc } from 'firebase/firestore';
import toast from 'react-hot-toast';

interface AuthModalProps {
  isOpen: boolean;
  onClose: () => void;
  defaultTab?: 'login' | 'signup' | 'vendor';
}

export default function AuthModal({ isOpen, onClose, defaultTab = 'login' }: AuthModalProps) {
  const [activeTab, setActiveTab] = useState(defaultTab);
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [name, setName] = useState('');
  const [shopName, setShopName] = useState('');
  const [category, setCategory] = useState('protein');
  const [isLoading, setIsLoading] = useState(false);

  const { loginWithGoogle } = useMarketplace();

  const handleGoogleLogin = async () => {
    setIsLoading(true);
    try {
      await loginWithGoogle();
      onClose();
    } catch (error) {
      console.error(error);
    } finally {
      setIsLoading(false);
    }
  };

  const handleManualAuth = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);

    try {
      if (activeTab === 'login') {
        if (!email.includes('@')) throw new Error('সঠিক ইমেইল এড্রেস প্রদান করুন।');
        const res = await signInWithEmailAndPassword(auth, email, password);
        toast.success(`স্বাগতম, ${res.user.displayName || 'ইউজার'}!`);
      } else {
        if (password.length < 6) throw new Error('পাসওয়ার্ড অন্তত ৬ অক্ষরের হতে হবে।');
        if (!name) throw new Error('আপনার পূর্ণ নাম প্রদান করুন।');
        
        const res = await createUserWithEmailAndPassword(auth, email, password);
        
        // Create profile
        const role = activeTab === 'vendor' ? 'vendor' : 'customer';
        const userData = {
          userId: res.user.uid,
          email: res.user.email || '',
          name: name,
          role: role,
          walletBalance: 0,
          createdAt: new Date().toISOString()
        };

        if (role === 'vendor') {
          if (!shopName) throw new Error('দোকানের নাম প্রদান করুন।');
          try {
            await setDoc(doc(db, 'vendors', res.user.uid), {
              vendorId: res.user.uid,
              storeName: shopName,
              category: category,
              verified: false,
              joinedAt: new Date().toISOString(),
              totalSales: 0,
              rating: 0
            });
          } catch (vendorError: any) {
            console.error("Vendor Profile Error:", vendorError);
            throw vendorError;
          }
        }

        try {
          await setDoc(doc(db, 'users', res.user.uid), userData);
        } catch (userError: any) {
          console.error("User Profile Error:", userError);
          throw userError;
        }
        
        toast.success(role === 'vendor' ? 'ভেন্ডর হিসেবে রেজিস্ট্রেশন সফল!' : 'রেজিস্ট্রেশন সফল হয়েছে!');
      }
      onClose();
    } catch (error: any) {
      console.error("Auth error:", error);
      if (error?.message?.includes('Quota exceeded') || error?.code === 'resource-exhausted') {
        toast.error('Firebase দৈনিক ফ্রি লিমিট শেষ হয়ে গেছে। দয়া করে আগামীকাল আবার চেষ্টা করুন।', {
          id: 'quota-error-auth',
          duration: 10000
        });
      } else {
        toast.error(error.message || 'অথেনটিকেশন সমস্যা হয়েছে।');
      }
    } finally {
      setIsLoading(false);
    }
  };

  if (!isOpen) return null;

  return (
    <div className="fixed inset-0 flex items-center justify-center z-[200] p-4">
      <motion.div 
        initial={{ opacity: 0 }} 
        animate={{ opacity: 1 }} 
        exit={{ opacity: 0 }}
        onClick={onClose}
        className="absolute inset-0 bg-slate-900/60 backdrop-blur-sm"
      />
      
      <motion.div
        initial={{ opacity: 0, scale: 0.9, y: 20 }}
        animate={{ opacity: 1, scale: 1, y: 0 }}
        exit={{ opacity: 0, scale: 0.9, y: 20 }}
        className="relative bg-white w-full max-w-lg rounded-[3rem] shadow-2xl overflow-hidden flex flex-col"
      >
        {/* Header Tabs */}
        <div className="flex bg-slate-50 p-2 m-6 rounded-2xl">
          {[
            { id: 'login', name: 'Log In', icon: LogIn },
            { id: 'signup', name: 'Register', icon: UserIcon },
            { id: 'vendor', name: 'Vendor', icon: StoreIcon },
          ].map((tab) => (
            <button
              key={tab.id}
              onClick={() => setActiveTab(tab.id as any)}
              className={`flex-1 flex items-center justify-center gap-2 py-3 rounded-xl text-sm font-bold transition-all ${
                activeTab === tab.id 
                  ? 'bg-white text-slate-900 shadow-md' 
                  : 'text-slate-400 hover:text-slate-600'
              }`}
            >
              <tab.icon className="w-4 h-4" />
              {tab.name}
            </button>
          ))}
        </div>

        <div className="px-10 pb-12 overflow-y-auto max-h-[80vh]">
          <div className="text-center mb-8">
            <h2 className="text-3xl font-black text-slate-900 tracking-tight">
              {activeTab === 'login' ? 'আবারও স্বাগতম!' : activeTab === 'vendor' ? 'ভেন্ডর হিসেবে যোগ দিন' : 'নতুন অ্যাকাউন্ট তৈরি করুন'}
            </h2>
            <p className="text-slate-500 mt-2 font-medium">
              {activeTab === 'login' 
                ? 'আপনার শপিং জার্নি শুরু করতে লগইন করুন' 
                : 'সবকিছু এখন এক ঠিকানায় – ফ্রেশ ও দ্রুত'}
            </p>
          </div>

          <form onSubmit={handleManualAuth} className="space-y-4">
            {activeTab !== 'login' && (
              <div className="space-y-2">
                <label className="text-xs font-bold text-slate-400 uppercase ml-1">Full Name</label>
                <div className="relative">
                  <User className="absolute left-4 top-4 w-5 h-5 text-slate-400" />
                  <input
                    required
                    type="text"
                    value={name}
                    onChange={(e) => setName(e.target.value)}
                    placeholder="আপনার নাম"
                    className="w-full bg-slate-50 border-none pl-12 pr-4 py-4 rounded-2xl focus:ring-2 focus:ring-amber-500/20 font-medium"
                  />
                </div>
              </div>
            )}

            {activeTab === 'vendor' && (
              <>
                <div className="space-y-2">
                  <label className="text-xs font-bold text-slate-400 uppercase ml-1">Shop Name</label>
                  <div className="relative">
                    <Store className="absolute left-4 top-4 w-5 h-5 text-slate-400" />
                    <input
                      required
                      type="text"
                      value={shopName}
                      onChange={(e) => setShopName(e.target.value)}
                      placeholder="আপনার দোকানের নাম"
                      className="w-full bg-slate-50 border-none pl-12 pr-4 py-4 rounded-2xl focus:ring-2 focus:ring-amber-500/20 font-medium"
                    />
                  </div>
                </div>
                <div className="space-y-2">
                  <label className="text-xs font-bold text-slate-400 uppercase ml-1">Product Category</label>
                  <div className="relative">
                    <Briefcase className="absolute left-4 top-4 w-5 h-5 text-slate-400" />
                    <select
                      value={category}
                      onChange={(e) => setCategory(e.target.value)}
                      className="w-full bg-slate-50 border-none pl-12 pr-4 py-4 rounded-2xl focus:ring-2 focus:ring-amber-500/20 font-medium appearance-none"
                    >
                      <option value="protein">মাছ ও মাংস (Protein)</option>
                      <option value="cosmetics">কসমেটিকস</option>
                      <option value="electronics">ইলেকট্রনিক্স</option>
                      <option value="fashion">ফ্যাশন</option>
                    </select>
                  </div>
                </div>
              </>
            )}

            <div className="space-y-2">
              <label className="text-xs font-bold text-slate-400 uppercase ml-1">Email Address</label>
              <div className="relative">
                <Mail className="absolute left-4 top-4 w-5 h-5 text-slate-400" />
                <input
                  required
                  type="email"
                  value={email}
                  onChange={(e) => setEmail(e.target.value)}
                  placeholder="name@example.com"
                  className="w-full bg-slate-50 border-none pl-12 pr-4 py-4 rounded-2xl focus:ring-2 focus:ring-amber-500/20 font-medium"
                />
              </div>
            </div>

            <div className="space-y-2">
              <label className="text-xs font-bold text-slate-400 uppercase ml-1">Password</label>
              <div className="relative">
                <Lock className="absolute left-4 top-4 w-5 h-5 text-slate-400" />
                <input
                  required
                  type="password"
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  placeholder="••••••••"
                  className="w-full bg-slate-50 border-none pl-12 pr-4 py-4 rounded-2xl focus:ring-2 focus:ring-amber-500/20 font-medium"
                />
              </div>
            </div>

            <button
              type="submit"
              disabled={isLoading}
              className="w-full bg-slate-900 text-white py-4 rounded-2xl font-bold hover:bg-amber-600 transition-all shadow-xl shadow-slate-200 flex items-center justify-center gap-2 group"
            >
              {isLoading ? 'Wait...' : (activeTab === 'login' ? 'লগইন করুন' : 'অ্যাকাউন্ট তৈরি করুন')}
              <ArrowRight className="w-5 h-5 group-hover:translate-x-1 transition-transform" />
            </button>
          </form>

          {activeTab !== 'vendor' && (
            <>
              <div className="relative my-8">
                <div className="absolute inset-0 flex items-center">
                  <div className="w-full border-t border-slate-100"></div>
                </div>
                <div className="relative flex justify-center text-xs uppercase">
                  <span className="bg-white px-4 text-slate-400 font-bold">অথবা লগইন করুন</span>
                </div>
              </div>

              <button
                onClick={handleGoogleLogin}
                className="w-full flex items-center justify-center gap-4 bg-white border-2 border-slate-100 py-4 rounded-2xl font-bold text-slate-600 hover:bg-slate-50 transition-all active:scale-95"
              >
                <img src="https://www.google.com/favicon.ico" alt="Google" className="w-5 h-5" />
                Google দিয়ে কন্টিনিউ করুন
              </button>
            </>
          )}

          <div className="mt-8 text-center text-sm">
            {activeTab === 'login' ? (
              <p className="text-slate-500">
                নতুন ইউজার?{' '}
                <button onClick={() => setActiveTab('signup')} className="text-amber-600 font-bold hover:underline">
                  এখান থেকে রেজিস্ট্রেশন করুন
                </button>
              </p>
            ) : (
              <p className="text-slate-500">
                ইতিমধ্যেই অ্যাকাউন্ট আছে?{' '}
                <button onClick={() => setActiveTab('login')} className="text-amber-600 font-bold hover:underline">
                  লগইন করুন
                </button>
              </p>
            )}
          </div>
        </div>

        <button 
          onClick={onClose}
          className="absolute top-6 right-6 p-2 bg-slate-100 text-slate-400 hover:text-slate-900 rounded-xl transition-colors"
        >
          <X className="w-5 h-5" />
        </button>
      </motion.div>
    </div>
  );
}
