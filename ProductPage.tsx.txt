import { useState, useEffect, FormEvent, useRef, ChangeEvent } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { ShoppingBag, Star, Zap, ChevronLeft, ChevronRight, ShieldCheck, Truck, RefreshCw, MessageSquare, Send, User, History, Share2, Leaf, Heart, Image as ImageIcon, X, Camera } from 'lucide-react';
import { useMarketplace } from '../context/MarketplaceContext';
import { collection, query, where, onSnapshot, addDoc, serverTimestamp, orderBy, getDoc, doc } from 'firebase/firestore';
import { ref, uploadBytes, getDownloadURL } from 'firebase/storage';
import { db, storage } from '../lib/firebase';
import { MOCK_PRODUCTS } from '../data/mockData';
import { Product } from '../types';
import toast from 'react-hot-toast';
import ChatSystem from './ChatSystem';

interface ProductReview {
  id: string;
  userId: string;
  userName: string;
  rating: number;
  comment: string;
  images?: string[];
  createdAt: any;
}

export default function ProductPage({ 
  product, 
  onBack, 
  onCheckout,
  onProductClick 
}: { 
  product: Product, 
  onBack: () => void, 
  onCheckout: () => void,
  onProductClick: (p: any) => void 
}) {
  const { addToCart, profile, user, products } = useMarketplace();
  const [activeImage, setActiveImage] = useState(0);
  const [showVideo, setShowVideo] = useState(false);
  const [quantity, setQuantity] = useState(1);
  const [reviews, setReviews] = useState<ProductReview[]>([]);
  const [recentlyViewed, sesetRecentlyViewedsesetRecentlyViewedtRecentlyViewedtRecentlyViewed] = useState<Product[]>([]);
  const [relatedProducts, setRelatedProducts] = useState<Product[]>([]);
  const [newReview, setNewReview] = useState({ rating: 5, comment: '' });
  const [reviewImages, setReviewImages] = useState<File[]>([]);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [showChat, setShowChat] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);
  const [vendorInfo, setVendorInfo] = useState<{
    name: string, 
    id: string, 
    verified?: boolean,
    logo?: string
  } | null>(null);

  const getCategoryLabel = (category: string, subCategory: string = '') => {
    switch (category) {
      case 'grocery':
        if (subCategory.includes('সবজি') || subCategory.includes('ফল')) {
          return { text: '১০০% ফ্রেশ', color: 'bg-green-100 text-green-700', icon: Leaf };
        }
        if (subCategory.includes('মাছ') || subCategory.includes('মাংস')) {
          return { text: 'ফ্রেশ ও অর্গানিক', color: 'bg-emerald-100 text-emerald-700', icon: Leaf };
        }
        return { text: 'অর্গানিক ফুড', color: 'bg-green-100 text-green-700', icon: Leaf };
      case 'pharmacy':
        return { text: 'অরিজিনাল ওষুধ', color: 'bg-blue-100 text-blue-700', icon: ShieldCheck };
      case 'electronics':
        return { text: 'অফিসিয়াল ওয়ারেন্টি', color: 'bg-cyan-100 text-cyan-700', icon: ShieldCheck };
      case 'wholesale':
        return { text: 'পাইকারি রেট', color: 'bg-amber-100 text-amber-700', icon: Zap };
      case 'fashion':
        return { text: 'প্রিমিয়াম ফ্যাশন', color: 'bg-purple-100 text-purple-700', icon: Star };
      case 'baby':
        return { text: 'নিরাপদ ও কোমল', color: 'bg-pink-100 text-pink-700', icon: Heart };
      default:
        return null;
    }
  };

  const badge = getCategoryLabel(product.category, product.subCategory);

  useEffect(() => {
    // Related Products Logic
    const filtered = (products.length > 0 ? products : MOCK_PRODUCTS)
      .filter(p => p.id !== product.id && (p.category === product.category || p.subCategory === product.subCategory))
      .slice(0, 10);
    setRelatedProducts(filtered as Product[]);
  }, [product.id, product.category, product.subCategory, products]);

  useEffect(() => {
    if (product.vendorId) {
      const fetchVendor = async () => {
        const snap = await getDoc(doc(db, 'vendors', product.vendorId!));
        if (snap.exists()) {
          const data = snap.data();
          setVendorInfo({ 
            name: data.storeName, 
            id: product.vendorId!,
            verified: data.verified,
            logo: data.storeLogo
          });
        }
      };
      fetchVendor();
    }
  }, [product.vendorId]);

  useEffect(() => {
    // Tracking Recently Viewed
    const viewed = JSON.parse(localStorage.getItem('recentlyViewed') || '[]');
    const updated = [product.id, ...viewed.filter((id: string) => id !== product.id)].slice(0, 10);
    localStorage.setItem('recentlyViewed', JSON.stringify(updated));

    // Load Recently Viewed Products (excluding current)
    const recentProducts = updated
      .filter((id: string) => id !== product.id)
      .map((id: string) => MOCK_PRODUCTS.find(p => p.id === id))
      .filter(Boolean) as unknown as Product[];
    setRecentlyViewed(recentProducts);

    const q = query(
      collection(db, 'reviews'),
      where('productId', '==', product.id),
      orderBy('createdAt', 'desc')
    );

    const unsubscribe = onSnapshot(q, (snapshot) => {
      const fetchedReviews = snapshot.docs.map(doc => ({
        id: doc.id,
        ...doc.data()
      })) as ProductReview[];
      setReviews(fetchedReviews);
    }, (err) => console.error("Reviews Listener Error:", err));

    return () => unsubscribe();
  }, [product.id]);

  const handleShare = () => {
    if (navigator.share) {
      navigator.share({
        title: product.name,
        text: `Check out this product: ${product.name}`,
        url: window.location.href,
      }).catch(console.error);
    } else {
      navigator.clipboard.writeText(window.location.href);
      toast.success('লিঙ্কটি কপি করা হয়েছে!');
    }
  };

  const handleAddToCart = () => {
    addToCart({
      productId: product.id,
      name: product.name,
      price: product.price,
      quantity: quantity,
      weight: 1
    });
    toast.success(`${product.name} ঝুড়িতে যোগ করা হয়েছে!`);
  };

  const handleImageChange = (e: ChangeEvent<HTMLInputElement>) => {
    if (e.target.files) {
      const files = Array.from(e.target.files);
      if (reviewImages.length + files.length > 5) {
        toast.error('সর্বোচ্চ ৫টি ছবি আপলোড করা যাবে');
        return;
      }
      setReviewImages([...reviewImages, ...files]);
    }
  };

  const removeImage = (index: number) => {
    setReviewImages(reviewImages.filter((_, i) => i !== index));
  };

  const handleSubmitReview = async (e: FormEvent) => {
    e.preventDefault();
    if (!user || !profile) {
      toast.error('রিভিউ দিতে দয়া করে লগইন করুন');
      return;
    }

    if (!newReview.comment.trim()) {
      toast.error('রিভিউ কমেন্ট লিখুন');
      return;
    }

    setIsSubmitting(true);
    try {
      let uploadedImageUrls: string[] = [];
      
      // Upload images to storage
      if (reviewImages.length > 0) {
        uploadedImageUrls = await Promise.all(
          reviewImages.map(async (file) => {
            const imageRef = ref(storage, `reviews/${product.id}/${Date.now()}-${file.name}`);
            const snapshot = await uploadBytes(imageRef, file);
            return await getDownloadURL(snapshot.ref);
          })
        );
      }

      await addDoc(collection(db, 'reviews'), {
        userId: user.uid,
        userName: profile.name,
        productId: product.id,
        rating: newReview.rating,
        comment: newReview.comment,
        images: uploadedImageUrls,
        createdAt: serverTimestamp()
      });
      setNewReview({ rating: 5, comment: '' });
      setReviewImages([]);
      toast.success('আপনার মূল্যবান রিভিউ যোগ করা হয়েছে!');
    } catch (error) {
      console.error('Error adding review:', error);
      toast.error('রিভিউ যোগ করতে সমস্যা হয়েছে');
    } finally {
      setIsSubmitting(false);
    }
  };

  const averageRating = reviews.length > 0 
    ? (reviews.reduce((acc, r) => acc + r.rating, 0) / reviews.length).toFixed(1)
    : product.rating;

  const images = product.images && product.images.length > 0 ? product.images : [product.image || 'https://via.placeholder.com/600'];

  return (
    <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-12">
      <button 
        onClick={onBack}
        className="flex items-center gap-2 text-slate-500 hover:text-amber-600 font-bold mb-8 transition-colors"
      >
        <ChevronLeft className="w-5 h-5" /> পেছনে ফিরে যান
      </button>

      <div className="grid grid-cols-1 lg:grid-cols-2 gap-16">
        {/* Left: Product Gallery */}
        <div className="space-y-6">
          <div className="relative aspect-square rounded-[3rem] overflow-hidden bg-slate-50 border border-slate-100 shadow-sm">
            <AnimatePresence mode="wait">
              {showVideo && product.video ? (
                <motion.div
                  key="video"
                  initial={{ opacity: 0 }}
                  animate={{ opacity: 1 }}
                  exit={{ opacity: 0 }}
                  className="w-full h-full bg-black flex items-center justify-center"
                >
                  <video 
                    src={product.video} 
                    className="w-full h-full object-contain"
                    controls
                    autoPlay
                  />
                  <button 
                    onClick={() => setShowVideo(false)}
                    className="absolute top-6 right-6 p-3 bg-white/20 backdrop-blur-md rounded-2xl text-white hover:bg-white hover:text-slate-900 transition-all font-black text-xs uppercase"
                  >
                    Close Video
                  </button>
                </motion.div>
              ) : (
                <motion.img
                  key={activeImage}
                  initial={{ opacity: 0, scale: 1.1 }}
                  animate={{ opacity: 1, scale: 1 }}
                  exit={{ opacity: 0, scale: 0.95 }}
                  transition={{ duration: 0.4, ease: [0.22, 1, 0.36, 1] }}
                  src={images[activeImage]}
                  alt={product.name}
                  className="w-full h-full object-cover"
                  referrerPolicy="no-referrer"
                />
              )}
            </AnimatePresence>

            {!showVideo && product.video && (
              <button 
                onClick={() => setShowVideo(true)}
                className="absolute top-6 right-6 px-4 py-2 bg-amber-500 text-white rounded-xl shadow-lg hover:scale-105 transition-transform flex items-center gap-2 font-bold text-xs uppercase"
              >
                <Zap className="w-4 h-4 fill-white" /> View Video
              </button>
            )}

            <div className="absolute inset-x-4 top-1/2 -translate-y-1/2 flex justify-between pointer-events-none">
              <button 
                onClick={() => setActiveImage((prev) => (prev === 0 ? images.length - 1 : prev - 1))}
                className="w-12 h-12 bg-white/90 backdrop-blur rounded-full flex items-center justify-center text-slate-900 shadow-xl pointer-events-auto hover:bg-amber-500 hover:text-white transition-all active:scale-90"
              >
                <ChevronLeft className="w-6 h-6" />
              </button>
              <button 
                onClick={() => setActiveImage((prev) => (prev === images.length - 1 ? 0 : prev + 1))}
                className="w-12 h-12 bg-white/90 backdrop-blur rounded-full flex items-center justify-center text-slate-900 shadow-xl pointer-events-auto hover:bg-amber-500 hover:text-white transition-all active:scale-90"
              >
                <ChevronRight className="w-6 h-6" />
              </button>
            </div>
          </div>

          <div className="flex gap-4 overflow-x-auto pb-2 no-scrollbar">
            {images.map((img, idx) => (
              <button
                key={idx}
                onClick={() => setActiveImage(idx)}
                className={`flex-shrink-0 w-24 h-24 rounded-2xl overflow-hidden border-2 transition-all ${
                  activeImage === idx ? 'border-amber-500 scale-105 shadow-lg' : 'border-transparent opacity-60 hover:opacity-100'
                }`}
              >
                <img src={img} alt="" className="w-full h-full object-cover" referrerPolicy="no-referrer" />
              </button>
            ))}
          </div>
        </div>

        {/* Right: Product Info */}
        <div className="flex flex-col">
          <div className="space-y-6 flex-1">
            <div className="space-y-2">
              <div className="flex items-center gap-2">
                <div className="text-amber-600 text-xs font-black uppercase tracking-[0.2em]">
                  {product.category}
                </div>
                {badge && (
                  <div className={`flex items-center gap-1 px-3 py-1 rounded-full text-[10px] font-black uppercase tracking-wider ${badge.color}`}>
                    <badge.icon className="w-3 h-3" />
                    {badge.text}
                  </div>
                )}
                <span className="w-4 h-px bg-amber-600 opacity-10"></span>
              </div>
              <h1 className="text-4xl sm:text-5xl font-black text-slate-900 leading-[1.1]">
                {product.name}
              </h1>
              <div className="flex items-center gap-4 pt-2">
                <div className="flex items-center gap-1.5 px-3 py-1 bg-amber-50 rounded-full text-amber-700 text-sm font-bold">
                  <Star className="w-4 h-4 fill-amber-500 text-amber-500" /> {averageRating} Rating
                </div>
                <div className="text-slate-400 text-sm font-medium">
                  {reviews.length} রিভিউ সম্পন্ন
                </div>
              </div>
            </div>

            <div className="flex items-baseline gap-4">
              <span className="text-5xl font-black text-slate-900 tracking-tight">৳{product.price}</span>
              <span className="text-slate-400 font-medium line-through">৳{Math.round(product.price * 1.2)}</span>
            </div>

            <p className="text-lg text-slate-500 leading-relaxed font-medium whitespace-pre-line">
              {product.description || `প্রিমিয়াম কোয়ালিটির এই ${product.name} আমরা আপনাদের জন্য নিয়ে এসেছি সরাসরি সোর্স থেকে। 
              এর বিশুদ্ধতা এবং গুণগত মান আমাদের অভিজ্ঞ টিম দ্বারা পরীক্ষিত।`}
            </p>

            <div className="space-y-4 pt-4">
              <div className="flex items-center gap-4">
                <div className="flex items-center bg-slate-50 rounded-2xl border border-slate-100 p-1">
                  <button 
                    onClick={() => setQuantity(Math.max(1, quantity - 1))}
                    className="w-10 h-10 flex items-center justify-center text-slate-900 font-bold hover:bg-white rounded-xl transition-all"
                  >
                    -
                  </button>
                  <span className="w-12 text-center font-black text-slate-900">{quantity}</span>
                  <button 
                    onClick={() => setQuantity(quantity + 1)}
                    className="w-10 h-10 flex items-center justify-center text-slate-900 font-bold hover:bg-white rounded-xl transition-all"
                  >
                    +
                  </button>
                </div>
                <div className={`text-sm font-bold ${product.stock > 0 ? 'text-green-600' : 'text-red-500'}`}>
                  {product.stock > 0 ? `শুধুমাত্র ${product.stock} ${product.unit} স্টকে আছে` : 'আউট অফ স্টক'}
                </div>
                
                <button 
                  onClick={handleShare}
                  className="ml-auto p-4 bg-slate-50 text-slate-400 hover:text-amber-600 rounded-2xl border border-slate-100 transition-all hover:scale-110"
                  title="Share"
                >
                  <Share2 className="w-5 h-5" />
                </button>
              </div>

              <div className="grid grid-cols-2 gap-4">
                <button 
                  onClick={handleAddToCart}
                  disabled={product.stock === 0}
                  className="py-5 bg-slate-100 text-slate-900 rounded-[2rem] font-black text-lg hover:bg-slate-200 transition-all active:scale-95 flex items-center justify-center gap-3 disabled:opacity-50"
                >
                  <ShoppingBag className="w-6 h-6" /> ঝুড়িতে
                </button>
                <button 
                  onClick={() => {
                    handleAddToCart();
                    onCheckout();
                  }}
                  disabled={product.stock === 0}
                  className="py-5 bg-slate-900 text-white rounded-[2rem] font-black text-lg hover:bg-amber-600 transition-all active:scale-95 flex items-center justify-center gap-3 shadow-2xl shadow-slate-200 disabled:opacity-50"
                >
                  কিনুন <Zap className="w-5 h-5 fill-amber-400 text-amber-400" />
                </button>
              </div>

              {product.vendorId && (
                <div className="pt-6">
                  <div className="flex items-center justify-between p-5 bg-slate-50 rounded-[2rem] border border-slate-100 group hover:border-amber-200 transition-all">
                    <div className="flex items-center gap-4">
                      <div className="relative w-14 h-14 bg-white rounded-2xl border border-slate-100 overflow-hidden shadow-sm group-hover:scale-110 transition-transform">
                        <img 
                          src={vendorInfo?.logo || `https://picsum.photos/seed/${product.vendorId}/200/200`} 
                          alt={vendorInfo?.name}
                          className="w-full h-full object-cover"
                          referrerPolicy="no-referrer"
                        />
                      </div>
                      <div>
                        <div className="flex items-center gap-2">
                          <h4 className="font-bold text-slate-900 leading-none">{vendorInfo?.name || 'মার্চেন্ট / সেলার'}</h4>
                          {vendorInfo?.verified && (
                            <div className="flex items-center justify-center p-0.5 bg-blue-500 text-white rounded-full shadow-lg shadow-blue-500/20" title="Verified Seller">
                              <ShieldCheck className="w-3.5 h-3.5" />
                            </div>
                          )}
                        </div>
                        <p className="text-[10px] font-black text-slate-400 uppercase tracking-widest mt-1">
                          {vendorInfo?.verified ? 'Verified Trusted Seller' : 'Platform Merchant'}
                        </p>
                      </div>
                    </div>
                    
                    <button 
                      onClick={() => {
                        if (!user) {
                          toast.error('মেসেজ করতে দয়া করে লগইন করুন');
                          return;
                        }
                        setShowChat(true);
                      }}
                      className="p-3 bg-white text-slate-900 rounded-xl border border-slate-100 shadow-sm hover:bg-slate-900 hover:text-white transition-all active:scale-95"
                      title="Message Seller"
                    >
                      <MessageSquare className="w-5 h-5" />
                    </button>
                  </div>
                </div>
              )}
            </div>
          </div>

          <div className="grid grid-cols-3 gap-6 pt-12 mt-12 border-t border-slate-50">
            {[
              { icon: ShieldCheck, label: '১০০% অরিজিনাল', sub: 'গ্যারান্টিড পণ্য' },
              { icon: Truck, label: 'দ্রুত ডেলিভারি', sub: '২৪-৪৮ ঘণ্টার মধ্যে' },
              { icon: RefreshCw, label: 'সহজ রিটার্ন', sub: '৩ দিনের মধ্যে' },
            ].map((feature, i) => (
              <div key={i} className="space-y-2">
                <div className="w-10 h-10 bg-slate-50 rounded-xl flex items-center justify-center text-amber-600 border border-slate-100">
                  <feature.icon className="w-5 h-5" />
                </div>
                <h4 className="font-black text-slate-900 text-sm uppercase tracking-tight">{feature.label}</h4>
                <p className="text-xs text-slate-400 font-medium">{feature.sub}</p>
              </div>
            ))}
          </div>
        </div>
      </div>

      {/* Reviews Section */}
      <div className="mt-24 grid grid-cols-1 lg:grid-cols-3 gap-16">
        <div className="lg:col-span-1 space-y-8">
          <div className="space-y-4">
            <h2 className="text-3xl font-black text-slate-900">ক্রেতাদের মতামত</h2>
            <div className="flex items-center gap-4">
              <div className="text-6xl font-black text-slate-900">{averageRating}</div>
              <div className="space-y-1">
                <div className="flex gap-0.5">
                  {[1, 2, 3, 4, 5].map((s) => (
                    <Star key={s} className={`w-5 h-5 ${s <= Math.round(Number(averageRating)) ? 'fill-amber-500 text-amber-500' : 'text-slate-200'}`} />
                  ))}
                </div>
                <p className="text-sm text-slate-400 font-bold uppercase tracking-widest">{reviews.length} মোট রিভিউ</p>
              </div>
            </div>
          </div>

          {/* Review Form */}
          <div className="bg-slate-50 rounded-[2.5rem] p-8 border border-slate-100 space-y-6">
            <div className="space-y-2">
              <h3 className="text-xl font-bold text-slate-900">আপনার মতামত দিন</h3>
              <p className="text-sm text-slate-500 font-medium">পণ্যের মান সম্পর্কে অন্য ক্রেতাদের জানাতে আপনার অভিজ্ঞতা শেয়ার করুন।</p>
            </div>

            <form onSubmit={handleSubmitReview} className="space-y-6">
              <div className="space-y-2">
                <label className="text-xs font-black text-slate-400 uppercase tracking-widest ml-1">রেটিং দিন</label>
                <div className="flex gap-2">
                  {[1, 2, 3, 4, 5].map((s) => (
                    <button
                      key={s}
                      type="button"
                      onClick={() => setNewReview({ ...newReview, rating: s })}
                      className={`w-10 h-10 rounded-xl flex items-center justify-center transition-all ${
                        newReview.rating >= s ? 'bg-amber-500 text-white' : 'bg-white text-slate-300 border border-slate-100'
                      }`}
                    >
                      <Star className={`w-5 h-5 ${newReview.rating >= s ? 'fill-current' : ''}`} />
                    </button>
                  ))}
                </div>
              </div>

              <div className="space-y-2">
                <label className="text-xs font-black text-slate-400 uppercase tracking-widest ml-1">আপনার কমেন্ট</label>
                <textarea
                  value={newReview.comment}
                  onChange={(e) => setNewReview({ ...newReview, comment: e.target.value })}
                  placeholder="পণ্যটি কেমন ছিল?"
                  rows={4}
                  className="w-full bg-white border border-slate-100 rounded-2xl p-4 text-sm focus:outline-none focus:ring-2 focus:ring-amber-500/20 transition-all"
                />
              </div>

              {/* Photo Upload Section */}
              <div className="space-y-2">
                <label className="text-xs font-black text-slate-400 uppercase tracking-widest ml-1">ছবি যোগ করুন (সর্বোচ্চ ৫টি)</label>
                <div className="flex flex-wrap gap-2">
                  {reviewImages.map((file, idx) => (
                    <div key={idx} className="relative w-16 h-16 rounded-xl overflow-hidden border border-slate-200">
                      <img src={URL.createObjectURL(file)} alt="" className="w-full h-full object-cover" />
                      <button 
                        type="button"
                        onClick={() => removeImage(idx)}
                        className="absolute top-0.5 right-0.5 bg-red-500 text-white p-0.5 rounded-full shadow-lg"
                      >
                        <X className="w-3 h-3" />
                      </button>
                    </div>
                  ))}
                  {reviewImages.length < 5 && (
                    <button
                      type="button"
                      onClick={() => fileInputRef.current?.click()}
                      className="w-16 h-16 rounded-xl border-2 border-dashed border-slate-200 flex flex-col items-center justify-center text-slate-400 hover:border-amber-500 hover:text-amber-500 transition-all"
                    >
                      <Camera className="w-5 h-5" />
                      <span className="text-[8px] font-black uppercase mt-1">Add</span>
                    </button>
                  )}
                </div>
                <input 
                  type="file" 
                  ref={fileInputRef} 
                  onChange={handleImageChange} 
                  accept="image/*" 
                  multiple 
                  className="hidden" 
                />
              </div>

              <button
                type="submit"
                disabled={isSubmitting}
                className="w-full bg-slate-900 text-white py-4 rounded-xl font-bold hover:bg-amber-600 transition-all flex items-center justify-center gap-2 disabled:opacity-50"
              >
                {isSubmitting ? 'প্রসেসিং হচ্ছে...' : (
                  <>সাবমিট করুন <Send className="w-4 h-4" /></>
                )}
              </button>
            </form>
          </div>
        </div>

        <div className="lg:col-span-2 space-y-6">
          <div className="flex items-center gap-4 mb-8">
            <MessageSquare className="w-6 h-6 text-amber-600" />
            <h3 className="text-2xl font-bold text-slate-900">সবগুলো রিভিউ ({reviews.length})</h3>
          </div>

          <div className="space-y-6">
            {reviews.length > 0 ? (
              reviews.map((review) => (
                <motion.div 
                  initial={{ opacity: 0, y: 20 }}
                  animate={{ opacity: 1, y: 0 }}
                  key={review.id} 
                  className="p-6 bg-white border border-slate-50 rounded-3xl space-y-4"
                >
                  <div className="flex items-center justify-between">
                    <div className="flex items-center gap-3">
                      <div className="w-10 h-10 bg-amber-100 rounded-full flex items-center justify-center text-amber-700 font-bold">
                        <User className="w-5 h-5" />
                      </div>
                      <div>
                        <h4 className="font-bold text-slate-900">{review.userName}</h4>
                        <div className="flex gap-0.5">
                          {[1, 2, 3, 4, 5].map((s) => (
                            <Star key={s} className={`w-3 h-3 ${s <= review.rating ? 'fill-amber-500 text-amber-500' : 'text-slate-100'}`} />
                          ))}
                        </div>
                      </div>
                    </div>
                    <span className="text-xs text-slate-400 font-medium">
                      {review.createdAt?.toDate ? review.createdAt.toDate().toLocaleDateString('bn-BD') : 'এখনই'}
                    </span>
                  </div>
                  <div className="space-y-3">
                    <p className="text-slate-600 text-sm leading-relaxed whitespace-pre-line px-1">
                      {review.comment}
                    </p>
                    {review.images && review.images.length > 0 && (
                      <div className="flex flex-wrap gap-2 px-1">
                        {review.images.map((img, i) => (
                          <div key={i} className="relative w-20 h-20 rounded-xl overflow-hidden border border-slate-100">
                             <img src={img} alt="Review" className="w-full h-full object-cover" referrerPolicy="no-referrer" />
                          </div>
                        ))}
                      </div>
                    )}
                  </div>
                </motion.div>
              ))
            ) : (
              <div className="py-20 text-center space-y-4 bg-slate-50 rounded-[3rem] border border-dashed border-slate-200">
                <div className="w-20 h-20 bg-white rounded-full flex items-center justify-center mx-auto shadow-sm">
                  <MessageSquare className="w-8 h-8 text-slate-200" />
                </div>
                <div className="space-y-1">
                  <h4 className="font-bold text-slate-900">এখনো কোনো রিভিউ নেই</h4>
                  <p className="text-sm text-slate-500">প্রথম ব্যক্তি হিসেবে আপনার মতামত শেয়ার করুন!</p>
                </div>
              </div>
            )}
          </div>
        </div>
      </div>

      {/* Related Products Carousel */}
      {relatedProducts.length > 0 && (
        <div className="mt-24 pt-24 border-t border-slate-100">
          <div className="flex items-center justify-between mb-12">
            <div className="flex items-center gap-3">
              <div className="p-2 bg-indigo-50 text-indigo-600 rounded-xl">
                <ShoppingBag className="w-6 h-6" />
              </div>
              <div>
                <h2 className="text-3xl font-black text-slate-900">সম্পর্কিত পণ্যসমূহ</h2>
                <p className="text-slate-500 font-medium">একই ক্যাটাগরির আরও কিছু পণ্য</p>
              </div>
            </div>
          </div>

          <div className="flex gap-6 overflow-x-auto pb-8 no-scrollbar scroll-smooth">
            {relatedProducts.map((rp) => (
              <motion.div
                key={rp.id}
                initial={{ opacity: 0, y: 20 }}
                whileInView={{ opacity: 1, y: 0 }}
                viewport={{ once: true }}
                onClick={() => onProductClick(rp)}
                className="flex-shrink-0 w-64 group bg-white rounded-[2rem] p-4 border border-slate-100 hover:border-amber-200 hover:shadow-2xl transition-all cursor-pointer"
              >
                <div className="relative aspect-square rounded-2xl overflow-hidden bg-slate-50 mb-3">
                  <img 
                    src={rp.image || (rp.images?.[0])} 
                    alt={rp.name}
                    className="w-full h-full object-cover group-hover:scale-110 transition-transform duration-500"
                    referrerPolicy="no-referrer"
                  />
                  {rp.stock < 5 && rp.stock > 0 && (
                    <div className="absolute top-2 right-2 bg-amber-500 text-white text-[10px] font-black px-2 py-0.5 rounded-full uppercase">
                      Stock Low
                    </div>
                  )}
                </div>
                <div>
                  <h4 className="font-bold text-slate-900 truncate group-hover:text-amber-600 transition-colors uppercase text-[10px] tracking-tight mb-1 opacity-60">
                    {rp.category}
                  </h4>
                  <h4 className="font-bold text-slate-900 truncate group-hover:text-amber-600 transition-colors text-sm">
                    {rp.name}
                  </h4>
                  <div className="flex items-center justify-between mt-2">
                    <p className="text-lg font-black text-slate-900">৳{rp.price}</p>
                    <div className="flex items-center gap-1 text-xs font-bold text-amber-500">
                      <Star className="w-3 h-3 fill-amber-500" /> {rp.rating || '4.0'}
                    </div>
                  </div>
                </div>
              </motion.div>
            ))}
          </div>
        </div>
      )}

      {/* Recently Viewed Section */}
      {recentlyViewed.length > 0 && (
        <div className="mt-24 pt-24 border-t border-slate-100">
          <div className="flex items-center gap-3 mb-12">
            <div className="p-2 bg-amber-50 text-amber-600 rounded-xl">
              <History className="w-6 h-6" />
            </div>
            <div>
              <h2 className="text-3xl font-black text-slate-900">সম্প্রতি দেখা পণ্যসমূহ</h2>
              <p className="text-slate-500 font-medium">আপনি সম্প্রতি এই আইটেমগুলো দেখেছেন</p>
            </div>
          </div>

          <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-5 gap-6">
            {recentlyViewed.map((rp) => (
              <motion.div
                key={rp.id}
                initial={{ opacity: 0, scale: 0.95 }}
                whileInView={{ opacity: 1, scale: 1 }}
                viewport={{ once: true }}
                onClick={() => onProductClick(rp)}
                className="group bg-white rounded-[2rem] p-4 border border-slate-100 hover:border-amber-200 hover:shadow-2xl transition-all cursor-pointer"
              >
                <div className="relative aspect-square rounded-2xl overflow-hidden bg-slate-50 mb-3">
                  <img 
                    src={rp.image} 
                    alt={rp.name}
                    className="w-full h-full object-cover group-hover:scale-110 transition-transform duration-500"
                    referrerPolicy="no-referrer"
                  />
                </div>
                <div>
                  <h4 className="font-bold text-slate-900 truncate group-hover:text-amber-600 transition-colors uppercase text-xs tracking-tight">
                    {rp.name}
                  </h4>
                  <p className="text-lg font-black text-slate-900 mt-1">৳{rp.price}</p>
                </div>
              </motion.div>
            ))}
          </div>
        </div>
      )}

      {/* Chat System Modal */}
      {product.vendorId && (
        <ChatSystem 
          isOpen={showChat}
          onClose={() => setShowChat(false)}
          receiverId={product.vendorId}
          receiverName={vendorInfo?.name || 'মার্চেন্ট / সেলার'}
          productId={product.id}
        />
      )}
    </div>
  );
}
