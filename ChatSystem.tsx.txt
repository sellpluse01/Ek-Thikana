import { useState, useEffect, useRef } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { Send, X, MessageCircle, User, Store, ShoppingBag } from 'lucide-react';
import { db, auth } from '../lib/firebase';
import { 
  collection, query, where, onSnapshot, addDoc, serverTimestamp, 
  orderBy, doc, setDoc, getDoc, updateDoc, limit, increment 
} from 'firebase/firestore';
import { useMarketplace } from '../context/MarketplaceContext';
import { Chat, ChatMessage } from '../types';
import toast from 'react-hot-toast';

interface ChatSystemProps {
  receiverId?: string;
  receiverName?: string;
  productId?: string;
  isOpen: boolean;
  onClose: () => void;
  inline?: boolean;
}

export default function ChatSystem({ receiverId, receiverName, productId, isOpen, onClose, inline = false }: ChatSystemProps) {
  const { user, profile } = useMarketplace();
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [newMessage, setNewMessage] = useState('');
  const [chatId, setChatId] = useState<string | null>(null);
  const [currentChat, setCurrentChat] = useState<Chat | null>(null);
  const [isChatReady, setIsChatReady] = useState(false);
  const scrollRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!user || !receiverId || !isOpen) return;

    // Sort IDs to create a consistent chat ID
    if (!receiverId || !user?.uid) return;
    const ids = [user.uid, receiverId].filter(Boolean).sort();
    const id = ids.join('_');
    setChatId(id);

    // Create or get chat document
    const setupChat = async () => {
      try {
        const chatRef = doc(db, 'chats', id);
        const chatSnap = await getDoc(chatRef);
        
        if (!chatSnap.exists()) {
          // Determine roles: if initiated from ProductPage (productId) or if current user is NOT vendor
          // but that's also not perfect. 
          // Let's use profile role but double check with receiverId
          const isActingAsVendor = profile?.role === 'vendor' && !productId;
          const finalCustomerId = isActingAsVendor ? receiverId : user.uid;
          const finalVendorId = isActingAsVendor ? user.uid : receiverId;

          if (!finalCustomerId || !finalVendorId) {
            console.error("Missing IDs for chat setup:", { finalCustomerId, finalVendorId });
            return;
          }

          const chatData: Omit<Chat, 'id'> = {
            participants: [user.uid, receiverId],
            customerId: finalCustomerId,
            vendorId: finalVendorId,
            customerName: isActingAsVendor ? (receiverName || 'Customer') : (profile?.name || 'Customer'),
            vendorName: isActingAsVendor ? (profile?.name || 'Vendor') : (receiverName || 'Vendor'),
            lastMessage: '',
            lastMessageTime: serverTimestamp(),
            unreadCountVendor: 0,
            unreadCountCustomer: 0,
            lastSenderId: ''
          };
          await setDoc(chatRef, chatData);
          setCurrentChat({ id: chatRef.id, ...chatData } as any);
          setIsChatReady(true);
        } else {
          const data = chatSnap.data() as Chat;
          setCurrentChat({ id: chatSnap.id, ...data });
          setIsChatReady(true);
          
          // Mark messages as read when opening chat
          const isVendorInThisChat = data.vendorId === user.uid;
          await updateDoc(chatRef, {
            [isVendorInThisChat ? 'unreadCountVendor' : 'unreadCountCustomer']: 0
          });
        }
      } catch (err) {
        console.error("Error setting up chat:", err);
      }
    };

    setupChat();

    // Listen for messages
    const q = query(
      collection(db, 'chats', id, 'messages'),
      orderBy('timestamp', 'asc'),
      limit(50)
    );

    const unsubscribe = onSnapshot(q, (snap) => {
      const msgs = snap.docs.map(d => ({ id: d.id, ...d.data() } as ChatMessage));
      setMessages(msgs);
      setTimeout(() => {
        scrollRef.current?.scrollIntoView({ behavior: 'smooth' });
      }, 100);
    }, (err) => console.error("Messages Listener Error:", err));

    return () => unsubscribe();
  }, [user, receiverId, isOpen]);

  const handleSendMessage = async (e?: any) => {
    if (e) e.preventDefault();
    if (!newMessage.trim() || !chatId || !user || !receiverId || !isChatReady || !currentChat) return;

    const msg = newMessage.trim();
    setNewMessage('');

    try {
      const isVendorInThisChat = currentChat.vendorId === user.uid;
      await addDoc(collection(db, 'chats', chatId, 'messages'), {
        senderId: user.uid,
        receiverId: receiverId,
        text: msg,
        timestamp: serverTimestamp(),
        productId: productId || currentChat.productId || null,
        read: false
      });

      await updateDoc(doc(db, 'chats', chatId), {
        lastMessage: msg,
        lastMessageTime: serverTimestamp(),
        lastSenderId: user.uid,
        [isVendorInThisChat ? 'unreadCountCustomer' : 'unreadCountVendor']: increment(1)
      });
    } catch (error) {
      console.error("Error sending message:", error);
      toast.error("মেসেজ পাঠানো যায়নি");
    }
  };

  if (!isOpen) return null;

  const content = (
    <div className={`${inline ? 'h-full flex flex-col w-full' : 'w-full max-w-lg bg-white rounded-[2.5rem] shadow-2xl overflow-hidden flex flex-col h-[80vh] sm:h-[600px] border border-slate-100'}`}>
      {!inline && (
        <div className="p-6 bg-slate-900 text-white flex items-center justify-between">
          <div className="flex items-center gap-4">
            <div className="w-12 h-12 bg-white/10 rounded-2xl flex items-center justify-center border border-white/10">
              <Store className="w-6 h-6 text-amber-400" />
            </div>
            <div>
              <h3 className="font-black text-lg tracking-tight">{receiverName || 'Seller'}</h3>
              <p className="text-[10px] text-amber-400 font-bold uppercase tracking-widest flex items-center gap-1">
                <span className="w-1.5 h-1.5 bg-green-500 rounded-full animate-pulse"></span> Personal Message
              </p>
            </div>
          </div>
          <button 
            onClick={onClose}
            className="w-10 h-10 bg-white/10 hover:bg-white/20 rounded-xl flex items-center justify-center transition-all"
          >
            <X className="w-5 h-5" />
          </button>
        </div>
      )}

      {/* Messages Area */}
      <div className="flex-1 overflow-y-auto p-6 space-y-4 bg-slate-50/50">
        {messages.length === 0 ? (
          <div className="h-full flex flex-col items-center justify-center text-center space-y-4 opacity-40">
            <div className="w-20 h-20 bg-slate-200 rounded-full flex items-center justify-center">
              <MessageCircle className="w-10 h-10" />
            </div>
            <div>
              <p className="font-bold text-slate-900">কথোপকথন শুরু করুন</p>
              <p className="text-sm">পণ্য সম্পর্কে জানতে মেসেজ দিন</p>
            </div>
          </div>
        ) : (
          messages.map((msg, i) => (
            <div 
              key={msg.id}
              className={`flex flex-col ${msg.senderId === user?.uid ? 'items-end' : 'items-start'}`}
            >
              <div 
                className={`max-w-[85%] p-4 rounded-3xl text-sm font-medium shadow-sm ${
                  msg.senderId === user?.uid 
                    ? 'bg-slate-900 text-white rounded-br-none' 
                    : 'bg-white text-slate-900 rounded-bl-none border border-slate-100'
                }`}
              >
                {msg.text}
              </div>
              <span className="text-[10px] text-slate-400 font-bold mt-1 px-1">
                {msg.timestamp?.toDate ? msg.timestamp.toDate().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }) : 'প্রেরিত'}
              </span>
            </div>
          ))
        )}
        <div ref={scrollRef} />
      </div>

      {/* Footer Input */}
      <div className="p-6 bg-white border-t border-slate-100">
        <form 
          onSubmit={handleSendMessage}
          className="relative flex items-center"
        >
          <input 
            type="text"
            value={newMessage}
            onChange={(e) => setNewMessage(e.target.value)}
            placeholder="আপনার মেসেজ লিখুন..."
            className="w-full bg-slate-50 border border-slate-100 rounded-2xl py-4 pl-6 pr-14 text-sm font-medium focus:outline-none focus:ring-2 focus:ring-amber-500/20 transition-all placeholder:text-slate-400"
          />
          <button 
            type="submit"
            className="absolute right-2 w-10 h-10 bg-slate-900 text-white rounded-xl flex items-center justify-center hover:bg-amber-600 transition-all active:scale-90"
          >
            <Send className="w-4 h-4" />
          </button>
        </form>
        <p className="text-center text-[10px] text-slate-400 font-bold mt-4 uppercase tracking-[0.2em]">
          Ek-Thikana Secure Messaging
        </p>
      </div>
    </div>
  );

  if (inline) return content;

  return (
    <div className="fixed inset-0 z-[100] flex items-end sm:items-center justify-center p-4 bg-slate-900/40 backdrop-blur-sm">
      <motion.div
        initial={{ opacity: 0, scale: 0.9, y: 20 }}
        animate={{ opacity: 1, scale: 1, y: 0 }}
        exit={{ opacity: 0, scale: 0.9, y: 20 }}
        className="contents"
      >
        {content}
      </motion.div>
    </div>
  );
}
