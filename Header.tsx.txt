import { useState, useEffect, useRef } from 'react';
import { Search, Wallet, Bell, ShoppingCart, User, LogOut, Loader2, Percent, MessageSquare } from 'lucide-react';
import { motion, AnimatePresence } from 'motion/react';
import { useMarketplace } from '../context/MarketplaceContext';
import { getSearchSuggestions } from '../services/aiService';
import { auth } from '../lib/firebase';
import { signInWithPopup, GoogleAuthProvider, signOut } from 'firebase/auth';
import toast from 'react-hot-toast';

export default function Header({ setView, onOpenCart, onOpenAuth, navigateToDashboard }: { 
  setView: (v: any) => void, 
  onOpenCart: () => void, 
  onOpenAuth: (tab?: any) => void,
  onCategorySelect?: (c: string) => void,
  navigateToDashboard?: (tab: string) => void
}) {
  const { profile, cart, user, logout, activeCoupons, unreadChatsCount, unreadNotificationsCount } = useMarketplace();
  const [searchQuery, setSearchQuery] = useState('');
  const [suggestions, setSuggestions] = useState<string[]>([]);
  const [isSearching, setIsSearching] = useState(false);
  const [showProfileMenu, setShowProfileMenu] = useState(false);
  const searchTimeout = useRef<NodeJS.Timeout | null>(null);

  const activePromo = activeCoupons?.[0];

  useEffect(() => {
    if (searchQuery.length > 2) {
      if (searchTimeout.current) clearTimeout(searchTimeout.current);
      searchTimeout.current = setTimeout(async () => {
        setIsSearching(true);
        const res = await getSearchSuggestions(searchQuery);
        setSuggestions(res);
        setIsSearching(false);
      }, 500);
    } else {
      setSuggestions([]);
    }
  }, [searchQuery]);

  const handleLogin = async () => {
    try {
      const provider = new GoogleAuthProvider();
      await signInWithPopup(auth, provider);
      toast.success('সফলভাবে লগইন করা হয়েছে!');
    } catch (error) {
      toast.error('লগইন করতে সমস্যা হয়েছে।');
    }
  };

  const handleLogout = async () => {
    await signOut(auth);
    toast.success('লগআউট করা হয়েছে।');
    setView('home');
  };

  return (
    <header className={`fixed top-0 left-0 right-0 ${activePromo ? 'h-28' : 'h-20'} bg-white/80 backdrop-blur-md border-b border-slate-100 z-50 transition-all duration-300`}>
      {/* Promo Ribbon */}
      {activePromo && (
        <div className="bg-slate-900 overflow-hidden relative group">
          <div className="max-w-7xl mx-auto py-2 px-4 flex items-center justify-center gap-3">
             <Percent className="w-3 h-3 text-amber-500 animate-bounce" />
             <p className="text-white text-[9px] sm:text-[10px] font-black uppercase tracking-[0.2em] flex items-center gap-2">
               Special Campaign: <span className="text-amber-500">"{activePromo.code}"</span> for {activePromo.discount}% OFF Sitewide!
             </p>
             <Percent className="w-3 h-3 text-amber-500 animate-bounce" />
          </div>
        </div>
      )}
      <div className={`max-w-7xl mx-auto px-4 ${activePromo ? 'h-20' : 'h-full'} flex items-center justify-between gap-2 sm:gap-8`}>
        {/* Logo */}
        <div 
          className="flex items-center gap-2 cursor-pointer group flex-shrink-0"
          onClick={() => setView('home')}
        >
          <div className="w-9 h-9 sm:w-10 sm:h-10 bg-amber-500 rounded-xl flex items-center justify-center text-white font-bold text-xl sm:text-2xl shadow-lg shadow-amber-200 group-hover:scale-105 transition-transform">
            E
          </div>
          <span className="text-lg sm:text-2xl font-bold bg-gradient-to-r from-slate-900 via-amber-950 to-amber-700 bg-clip-text text-transparent block">
            Ek-Thikana
          </span>
        </div>

        {/* Search Bar */}
        <div className="flex-1 max-w-xl relative">
          <div className="relative group">
            <input
              type="text"
              placeholder="AI Search..."
              className="w-full h-9 sm:h-11 pl-9 sm:pl-12 pr-4 bg-slate-50 border border-slate-200 rounded-full focus:outline-none focus:ring-2 focus:ring-amber-500/20 focus:border-amber-500 transition-all text-sm"
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
            />
            <Search className="absolute left-3 sm:left-4 top-2.5 sm:top-3 text-slate-400 w-4 h-4 sm:w-5 sm:h-5 group-focus-within:text-amber-500 transition-colors" />
            
            {isSearching && (
              <Loader2 className="absolute right-3 sm:right-4 top-2.5 sm:top-3 text-amber-500 w-4 h-4 sm:w-5 sm:h-5 animate-spin" />
            )}
          </div>

          {/* AI Suggestions */}
          <AnimatePresence>
            {suggestions.length > 0 && (
              <motion.div
                initial={{ opacity: 0, y: 10 }}
                animate={{ opacity: 1, y: 0 }}
                exit={{ opacity: 0, y: 10 }}
                className="absolute top-14 left-0 right-0 bg-white border border-slate-100 rounded-2xl shadow-xl p-2 z-50 overflow-hidden"
              >
                {suggestions.map((s, i) => (
                  <div 
                    key={i}
                    className="p-3 hover:bg-slate-50 rounded-xl cursor-pointer flex items-center gap-3 text-slate-700"
                    onClick={() => {
                      setSearchQuery(s);
                      setSuggestions([]);
                    }}
                  >
                    <Search className="w-4 h-4 text-slate-400" />
                    {s}
                  </div>
                ))}
              </motion.div>
            )}
          </AnimatePresence>
        </div>

        {/* Right Nav */}
        <div className="flex items-center gap-4 sm:gap-6">
          {!user ? (
            <button 
              onClick={() => onOpenAuth('vendor')}
              className="hidden lg:flex items-center gap-2 text-sm font-bold text-amber-600 hover:text-amber-700 transition-colors"
            >
              Become a Vendor
            </button>
          ) : profile?.role === 'customer' ? (
            <button 
              onClick={() => setView('dashboard')}
              className="hidden lg:flex items-center gap-2 text-sm font-bold text-amber-600 hover:text-amber-700 transition-colors"
            >
              Start Selling
            </button>
          ) : null}

          {user ? (
            <>
              {/* Wallet */}
              <div className="hidden lg:flex items-center gap-2 bg-amber-50 px-4 py-2 rounded-full border border-amber-100">
                <Wallet className="w-4 h-4 text-amber-600" />
                <span className="font-bold text-amber-900">৳{profile?.walletBalance?.toLocaleString() || 0}</span>
              </div>

              {/* Icons */}
              <div className="flex items-center gap-2 sm:gap-4">
                <button 
                  onClick={() => navigateToDashboard?.('messages')}
                  className="p-2 text-slate-600 hover:bg-slate-100 rounded-full relative transition-colors"
                >
                  <MessageSquare className="w-6 h-6" />
                  {unreadChatsCount > 0 && (
                    <span className="absolute -top-1 -right-1 bg-red-600 text-white text-[10px] font-bold w-5 h-5 flex items-center justify-center rounded-full border-2 border-white">
                      {unreadChatsCount}
                    </span>
                  )}
                </button>

                <button 
                  onClick={() => navigateToDashboard?.('notifications')}
                  className="p-2 text-slate-600 hover:bg-slate-100 rounded-full relative transition-colors"
                >
                  <Bell className="w-6 h-6" />
                  {unreadNotificationsCount > 0 && (
                    <span className="absolute -top-1 -right-1 bg-red-600 text-white text-[10px] font-bold w-5 h-5 flex items-center justify-center rounded-full border-2 border-white">
                      {unreadNotificationsCount}
                    </span>
                  )}
                </button>
                
                <button 
                  onClick={onOpenCart}
                  className="p-2 text-slate-600 hover:bg-slate-100 rounded-full relative transition-colors"
                >
                  <ShoppingCart className="w-6 h-6" />
                  {cart.length > 0 && (
                    <span className="absolute -top-1 -right-1 bg-amber-600 text-white text-[10px] font-bold w-5 h-5 flex items-center justify-center rounded-full border-2 border-white">
                      {cart.length}
                    </span>
                  )}
                </button>

                {/* Profile */}
                <div className="relative">
                  <button 
                    onClick={() => setShowProfileMenu(!showProfileMenu)}
                    className="flex items-center gap-2 p-1 hover:bg-slate-100 rounded-full transition-colors"
                  >
                    {profile?.profilePic ? (
                      <img src={profile.profilePic} alt="P" className="w-8 h-8 rounded-full border border-slate-200" referrerPolicy="no-referrer" />
                    ) : (
                      <div className="w-8 h-8 bg-amber-100 text-amber-600 rounded-full flex items-center justify-center">
                        <User className="w-5 h-5" />
                      </div>
                    )}
                  </button>

                  <AnimatePresence>
                    {showProfileMenu && (
                      <motion.div
                        initial={{ opacity: 0, scale: 0.95, y: 10 }}
                        animate={{ opacity: 1, scale: 1, y: 0 }}
                        exit={{ opacity: 0, scale: 0.95, y: 10 }}
                        className="absolute right-0 top-12 w-48 bg-white border border-slate-100 rounded-2xl shadow-xl overflow-hidden"
                      >
                        <div className="p-4 border-b border-slate-50">
                          <p className="font-bold text-slate-900 truncate">{profile?.name}</p>
                          <p className="text-xs text-slate-500 truncate capitalize">{profile?.role}</p>
                        </div>
                        <button 
                          onClick={() => { setView('dashboard'); setShowProfileMenu(false); }}
                          className="w-full flex items-center gap-3 p-3 text-slate-700 hover:bg-slate-50 transition-colors"
                        >
                          <User className="w-4 h-4" /> Dashboard
                        </button>
                        <button 
                          onClick={handleLogout}
                          className="w-full flex items-center gap-3 p-3 text-red-600 hover:bg-red-50 transition-colors"
                        >
                          <LogOut className="w-4 h-4" /> Logout
                        </button>
                      </motion.div>
                    )}
                  </AnimatePresence>
                </div>
              </div>
            </>
          ) : (
            <button 
              onClick={() => onOpenAuth('login')}
              className="bg-slate-900 text-white px-6 py-2.5 rounded-full font-medium hover:bg-slate-800 transition-colors active:scale-95 shadow-lg shadow-slate-200"
            >
              Sign In
            </button>
          )}
        </div>
      </div>
    </header>
  );
}
